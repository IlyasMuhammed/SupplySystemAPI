import { DataSource, QueryFailedError, Repository } from 'typeorm';
import { User } from '../entities/User.entity';
import { PasswordService } from '../services/PasswordService';
 
export interface CreateUserDto {
  email: string;
  password: string;
  firstName: string;
  lastName?: string;
  phone?: string;
  address?: string;
}
 
export class DuplicateEmailError extends Error {
  constructor(email: string) {
    super(`A user with email '${email}' already exists.`);
    this.name = 'DuplicateEmailError';
  }
}
 
export class UserRepository {
  private readonly repo: Repository<User>;
 
  constructor(
    private readonly dataSource: DataSource,
    private readonly passwordService: PasswordService,
  ) {
    this.repo = dataSource.getRepository(User);
  }
 
  /**
   * Find a user by their email address (case-insensitive).
   * @returns The User entity or null if not found.
   */
  async findByEmail(email: string): Promise<User | null> {
    if (!email?.trim()) return null;
    return this.repo.findOne({
      where: { email: email.toLowerCase().trim(), isDeleted: false },
      relations: ['role'],
    });
  }
 
  /**
   * Find a user by their UUID primary key.
   * @returns The User entity or null if not found.
   */
  async findById(id: string): Promise<User | null> {
    if (!id?.trim()) return null;
    return this.repo.findOne({
      where: { id, isDeleted: false },
      relations: ['role'],
    });
  }
 
  /**
   * Create and persist a new user.
   * The password is hashed before being stored — plain text is NEVER saved.
   *
   * @throws DuplicateEmailError when the email unique constraint is violated.
   */
  async create(dto: CreateUserDto): Promise<User> {
    const passwordHash = await this.passwordService.hash(dto.password);
 
    const user = this.repo.create({
      email: dto.email.toLowerCase().trim(),
      passwordHash,
      firstName: dto.firstName,
      lastName: dto.lastName,
      phone: dto.phone,
      address: dto.address,
      isActive: false,
      isDeleted: false,
    });
 
    try {
      return await this.repo.save(user);
    } catch (err) {
      if (
        err instanceof QueryFailedError &&
        (err as QueryFailedError & { code?: string }).code === '23505'
      ) {
        throw new DuplicateEmailError(dto.email);
      }
      throw err;
    }
  }
 
  /**
   * Replace a user's password hash.
   * The new password is hashed before being stored — plain text is NEVER saved.
   *
   * @returns true if the user was found and updated, false otherwise.
   */
  async updatePassword(userId: string, newPassword: string): Promise<boolean> {
    const user = await this.repo.findOneBy({ id: userId, isDeleted: false });
    if (!user) return false;
 
    user.passwordHash = await this.passwordService.hash(newPassword);
    await this.repo.save(user);
    return true;
  }
}
